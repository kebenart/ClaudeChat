- [ ] **Step 1: 实现 `IOSAppModel.swift`**

```swift
import Foundation
import SwiftData
import ChatKit

@MainActor
@Observable
public final class IOSAppModel {
    public var conversations: [ImConversationDTO] = []
    public var unread: [String: Int] = [:]

    private let storage: Storage
    private let engine: ImSyncEngine
    private let api: APIClient
    private let socket: ChatSocket
    private let deviceId: String

    public init(baseURL: URL, token: String?) {
        let container = (try? StorageContainer.makeOnDisk(profileId: UUID(uuidString: "00000000-0000-0000-0000-000000000001")!))
            ?? (try! StorageContainer.makeInMemory())
        self.storage = Storage(container: container)
        self.engine = ImSyncEngine(storage: storage)
        self.api = APIClient(baseURL: baseURL)
        self.socket = ChatSocket()
        self.deviceId = DeviceIdentity.current()
        Task { await api.setToken(token) }
    }

    /// Initial sync + connect WS + consume im:* frames.
    public func start(baseURL: URL, token: String) async {
        await api.setBaseURL(baseURL)
        await api.setToken(token)
        await pagedSync()
        await refresh()
        do {
            try await socket.connect(baseURL: baseURL, token: token)
            Task { await consume(socket.events) }
        } catch {
            print("[IOSAppModel] socket connect failed: \(error)")
        }
    }

    private func pagedSync() async {
        var since = await storage.imSyncCursor()
        while true {
            do {
                let resp = try await api.fetchImSync(since: since)
                await engine.applySync(resp)
                if !resp.hasMore || resp.cursor <= since { break }
                since = resp.cursor
            } catch {
                print("[IOSAppModel] sync failed: \(error)"); break
            }
        }
    }

    private func consume(_ stream: AsyncStream<ServerEvent>) async {
        for await event in stream {
            switch event {
            case .imPoke:
                await pagedSync()
            case .imMessage, .imRead:
                await engine.applyFrame(event)
            default:
                break
            }
            await refresh()
        }
    }

    public func refresh() async {
        var convs = await storage.imConversations()
        convs.sort { ($0.isPinned ? 1 : 0, $0.lastActivityAt) > ($1.isPinned ? 1 : 0, $1.lastActivityAt) }
        let u = await engine.computeUnread()
        self.conversations = convs
        self.unread = u
    }

    public func messages(_ conversationId: String) async -> [ImMessageDTO] {
        await storage.imMessages(conversationId: conversationId)
    }

    public func markRead(_ conversationId: String) async {
        let seq = conversations.first { $0.id == conversationId }?.lastSeq ?? 0
        await storage.setImReadCursor(conversationId: conversationId, deviceId: deviceId, lastReadSeq: seq)
        await refresh()
        try? await api.postImRead(conversationId: conversationId, deviceId: deviceId, lastReadSeq: seq)
    }
}
```

> **校验**:`ChatSocket()` 的无参 init 与 `connect(baseURL:token:)`、`events` 属性以 `Protocols.swift` `ChatSocketProtocol` 为准;若 `ChatSocket` init 需参数,按真实签名调整。`StorageContainer.makeOnDisk(profileId:)` 已存在。Task 1 拆分后这些核心类型均 `public` 且 iOS 可用。

import SwiftUI
import ChatKit

struct ChatDetailView: View {
    let conversation: ImConversationDTO
    @Environment(IOSAppModel.self) private var model
    @State private var messages: [ImMessageDTO] = []

    var body: some View {
        ScrollView {
            LazyVStack(alignment: .leading, spacing: 10) {
                ForEach(messages) { m in
                    bubble(m)
                }
            }
            .padding(12)
        }
        .navigationTitle(conversation.title ?? "对话")
        .navigationBarTitleDisplayMode(.inline)
        .task {
            messages = await model.messages(conversation.id)
            await model.markRead(conversation.id)
        }
        // Reload when new messages arrive (model.conversations bumps on refresh).
        .onChange(of: model.conversations.first(where: { $0.id == conversation.id })?.lastSeq) { _, _ in
            Task {
                messages = await model.messages(conversation.id)
                await model.markRead(conversation.id)
            }
        }
    }

    @ViewBuilder private func bubble(_ m: ImMessageDTO) -> some View {
        let isUser = m.role == "user"
        VStack(alignment: isUser ? .trailing : .leading, spacing: 2) {
            HStack {
                if isUser { Spacer(minLength: 40) }
                Text(m.content)
                    .padding(10)
                    .background(isUser ? Color.green.opacity(0.85) : Color(.secondarySystemBackground))
                    .foregroundStyle(isUser ? .white : .primary)
                    .clipShape(RoundedRectangle(cornerRadius: 10))
                if !isUser { Spacer(minLength: 40) }
            }
            if let trace = m.toolTrace, trace.count > 0 {
                Text("执行了 \(trace.count) 个操作")
                    .font(.system(size: 11)).foregroundStyle(.secondary)
                    .frame(maxWidth: .infinity, alignment: isUser ? .trailing : .leading)
            }
        }
        .frame(maxWidth: .infinity, alignment: isUser ? .trailing : .leading)
    }
}

```swift
import Foundation
import UserNotifications
import ChatKit

@MainActor
public enum IOSNotifications {
    public static func requestAuthorization() {
        UNUserNotificationCenter.current().requestAuthorization(options: [.alert, .sound, .badge]) { _, _ in }
    }

    /// Fire a local notification for an assistant reply + set the app icon badge.
    public static func notifyAssistantReply(conversationTitle: String, preview: String, totalUnread: Int) {
        let content = UNMutableNotificationContent()
        content.title = "Claude 在「\(conversationTitle)」中回复了"
        content.body = preview
        content.sound = .default
        content.badge = NSNumber(value: totalUnread)
        let req = UNNotificationRequest(identifier: UUID().uuidString, content: content, trigger: nil)
        UNUserNotificationCenter.current().add(req)
    }

    public static func setBadge(_ count: Int) {
        UNUserNotificationCenter.current().setBadgeCount(count)
    }
}
```

    private func consume(_ stream: AsyncStream<ServerEvent>) async {
        for await event in stream {
            switch event {
            case .imPoke:
                await pagedSync()
            case let .imMessage(conversationId, message):
                await engine.applyFrame(event)

                if message.role == "assistant", conversationId != foregroundConversationId {

                await engine.applyFrame(event)

            }

        }
    }
